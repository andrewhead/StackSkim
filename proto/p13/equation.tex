\end{align*}

\begin{center}
where
\end{center}
\begin{align*}
    \text{Cost}(c) = \text{dist}(c_f, c_s) + \text{size}(c_f) + \text{size}(c_s) \\
    - \min(\text{size}(c) - \text{limit\_size}, 0) \\
    + \Sigma_{i=0}^V \left| \text{avg} (\text{values} (c, i)) - 0.5 \right|
\end{align*}
